import { useState } from "react";
export const ObjectiveQuiz = ({options}) => {
  const Checkbox = ({option}) => {
    const { name, image, value } = option;
    return (
      <label className="cu_label">
        <input type="checkbox" value={value} />
        { name || `` }
        { image && (<img src={image} alt="" />) }
      </label>
    )
  };
  return (
    <>
      {options.map((option, index) => (<Checkbox option={option} key={index}></Checkbox>))}
    </>
  );
};
